1. document.getElementById();

Gibt ein Element anhand seiner ID zurück.

<div id="triangle">Ich bin ein Dreieck</div>
document.getElementById("triangle");


      
      

    


    


    
    

      
2. document.getElementsByClassName()

      
Gibt alle Elemente mit der angegebenen Klasse in Form eines Arrays (bzw. einer HTMLCollection) zurück.

      
<div class="rectangle">Ich bin ein Rechteck</div>
<div class="rectangle">Ich bin auch ein Rechteck</div>
<div class="rectangle">Ich bin erst recht ein Rechteck</div>


document.getElementsByClassName("rectangle");

  
  

  



const rectangles = document.getElementsByClassName("rectangle");
  
rectangles.addEventListener("click", windowAlert)

function windowAlert() {
  alert("Hallo Welt")
}





  

  

for (let i = 0; i < rectangles.length; i++;) {
  console.log(rectangles[i])

  rectangles[i].addEventListener("click", windowAlert);
}

  
  
  
  
  
        

        

        

    


    


    
    

      
3. document.getElementsByTagName()

      
Gibt alle Elemente mit dem angegebenen TagName in Form eines Arrays (bzw. einer HTMLCollection) genau wie bei document.getElementsByClassName(); zurück.

      
<pentagon></pentagon>

      
document.getElementsByTagName("pentagon");

      
      

      


      
Anstatt über alle Elemente im Array zu iterieren kann man auch einfach den Index des Elements im Array angeben. Dies ist aber nur zu empfehlen wenn es nur wenige Elemente mit dem entsprechenden TagName gibt.

      
document.getElementsByTagName("pentagon[0]");

      
      

      
      
    


    


    
    

      
4. document.querySelector()

      
Hiermit kann man sowohl eine Klasse, eine ID, als auch einen Tag greifen. Hierbei ist zu beachten dass man demnach natürlich 
      
<div class="hexagon">Ich bin ein Hexagon</div>

      
document.querySelector(".hexagon");

      
      

      


      
<div id="hexagon">Ich bin auch ein Hexagon</div>

      
document.querySelector("#hexagon");

      
      

      


      
<hexagon>Ich bin Hexagon Nummer 3</hexagon>

      
document.querySelector("hexagon");

      
      

      





      
<div class="container">
  <div class="circle"></div>
</div>

      
document.querySelector(".container .circle");

      
      

      



const container = document.querySelector(".container");

container.querySelector(".circle");

<septagon>Ich bin ein Septagon 1</septagon>
